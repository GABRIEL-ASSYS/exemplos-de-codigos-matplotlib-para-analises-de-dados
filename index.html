import matplotlib.pyplot as plt
from mpl_toolkits.axes_grid1.inset_locator import inset_axes
from matplotlib.patches import ConnectionPatch
import matplotlib.gridspec as gridspec
import numpy as np

# Dados de exemplo
x = np.linspace(0, 10, 100)
y1 = np.sin(x)
y2 = np.cos(x)

# 1. Eixos secund√°rios
fig, ax1 = plt.subplots(figsize=(10, 6), dpi=100)
ax1.plot(x, y1, color='b', label='Seno')
ax1.set_xlabel('X')
ax1.set_ylabel('sin(x)', color='b')
ax1.tick_params(axis='y', labelcolor='b')
ax2 = ax1.twinx()
ax2.plot(x, y2, color='r', linestyle='--', label='Cosseno')
ax2.set_ylabel('cos(x)', color='r')
ax2.tick_params(axis='y', labelcolor='r')

# 2. Inset plot (zoom no seno)
axins = inset_axes(ax1, width="30%", height="30%", loc='upper right')
axins.plot(x, y1, color='green')
axins.set_title('Zoom Seno')

# 3. Broken axis com gridspec
fig2 = plt.figure(figsize=(6, 4), dpi=100)
gs = gridspec.GridSpec(2, 1, height_ratios=[1, 2])
ax_top = fig2.add_subplot(gs[0])
ax_bottom = fig2.add_subplot(gs[1], sharex=ax_top)
ax_top.plot(x, y1)
ax_bottom.plot(x, y1)
ax_top.spines['bottom'].set_visible(False)
ax_bottom.spines['top'].set_visible(False)
ax_top.tick_params(labelbottom=False)
con = ConnectionPatch(
    xyA=(0,1), xyB=(0,0),
    coordsA="axes fraction", coordsB="axes fraction",
    axesA=ax_top, axesB=ax_bottom,
    arrowstyle='-'
)
ax_bottom.add_artist(con)

# 4. Textos em LaTeX
plt.rc('text', usetex=True)
fig3, ax3 = plt.subplots(figsize=(6, 3), dpi=100)
ax3.set_title(r'$\alpha > \beta$', fontsize=14)
ax3.plot(x, y1, label=r'$\sin(x)$')
ax3.legend()

plt.show()
